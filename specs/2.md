Bisa, dan **paling rapi** kalau JWT ditaruh sebagai **module `auth` + middleware global**, bukan dicampur ke module `user`.

Untuk struktur Anda, saya sarankan flow seperti ini:

```text id="7c36h1"
login handler
  → auth service
    → user repository
    → password verify
    → token generator (JWT)
    → refresh token store (Redis / DB)
  → response access_token + refresh_token

request ke endpoint private
  → JWT middleware
    → parse token
    → validate signature + exp + nbf + iss + aud
    → inject claims ke context
  → handler business
```

Library Go yang umum dipakai untuk JWT adalah `github.com/golang-jwt/jwt/v5`, dan versi v5 memakai `RegisteredClaims` sebagai standar claims type. ([Go Packages][1])

# Pola JWT yang saya sarankan

Untuk production API seperti punya Anda, jangan hanya 1 token. Pakai:

* **access token**: short-lived, mis. 15 menit
* **refresh token**: lebih panjang, mis. 7 hari
* refresh token disimpan di Redis agar bisa revoke/logout semua device

Kenapa?

* access token pendek lebih aman
* refresh token bisa di-rotate
* cocok untuk stateless API node

Kalau API Anda horizontal scale, validasi access token tetap stateless, sedangkan kontrol sesi ada di Redis. Ini cocok dengan Redis sebagai shared store dan middleware/rate limit terdistribusi. ([GitHub][2])

# Token service implementation

```go id="xkc19i"
// File: internal/modules/auth/service/service.go
package service

import (
	"context"
	"crypto/rand"
	"encoding/hex"
	"errors"
	"time"

	"github.com/golang-jwt/jwt/v5"
	"yourapp/internal/config"
	authdomain "yourapp/internal/modules/auth/domain"
)

type RefreshStore interface {
	Save(ctx context.Context, sessionID string, userID int64, ttl time.Duration) error
	Exists(ctx context.Context, sessionID string) (bool, error)
	Delete(ctx context.Context, sessionID string) error
}

type PasswordVerifier interface {
	Verify(hashedPassword, plainPassword string) bool
}

type UserAuthRepository interface {
	FindAuthByEmail(ctx context.Context, email string) (*UserAuth, error)
}

type UserAuth struct {
	ID             int64
	Email          string
	PasswordHash   string
	Role           string
	IsActive       bool
}

type Service struct {
	cfg           config.JWTConfig
	userRepo      UserAuthRepository
	refreshStore  RefreshStore
	passwordCheck PasswordVerifier
}

func New(
	cfg config.JWTConfig,
	userRepo UserAuthRepository,
	refreshStore RefreshStore,
	passwordCheck PasswordVerifier,
) *Service {
	return &Service{
		cfg:           cfg,
		userRepo:      userRepo,
		refreshStore:  refreshStore,
		passwordCheck: passwordCheck,
	}
}

func (s *Service) Login(ctx context.Context, req authdomain.LoginInput) (*authdomain.TokenPair, error) {
	user, err := s.userRepo.FindAuthByEmail(ctx, req.Email)
	if err != nil {
		return nil, err
	}
	if user == nil || !user.IsActive {
		return nil, errors.New("invalid credentials")
	}

	if !s.passwordCheck.Verify(user.PasswordHash, req.Password) {
		return nil, errors.New("invalid credentials")
	}

	return s.generateTokenPair(ctx, user.ID, user.Email, user.Role)
}

func (s *Service) Refresh(ctx context.Context, refreshToken string) (*authdomain.TokenPair, error) {
	claims := &authdomain.RefreshClaims{}

	token, err := jwt.ParseWithClaims(refreshToken, claims, func(token *jwt.Token) (any, error) {
		return []byte(s.cfg.RefreshSecret), nil
	}, jwt.WithAudience(s.cfg.Audience), jwt.WithIssuer(s.cfg.Issuer))
	if err != nil || !token.Valid {
		return nil, errors.New("invalid refresh token")
	}

	if claims.Type != "refresh" {
		return nil, errors.New("invalid token type")
	}

	exists, err := s.refreshStore.Exists(ctx, claims.SessionID)
	if err != nil {
		return nil, err
	}
	if !exists {
		return nil, errors.New("refresh session revoked")
	}

	_ = s.refreshStore.Delete(ctx, claims.SessionID)

	return s.generateTokenPair(ctx, claims.UserID, "", "")
}

func (s *Service) ParseAccessToken(tokenStr string) (*authdomain.AccessClaims, error) {
	claims := &authdomain.AccessClaims{}

	token, err := jwt.ParseWithClaims(tokenStr, claims, func(token *jwt.Token) (any, error) {
		return []byte(s.cfg.AccessSecret), nil
	}, jwt.WithAudience(s.cfg.Audience), jwt.WithIssuer(s.cfg.Issuer))
	if err != nil || !token.Valid {
		return nil, errors.New("invalid access token")
	}

	if claims.Type != "access" {
		return nil, errors.New("invalid token type")
	}

	return claims, nil
}

func (s *Service) Logout(ctx context.Context, refreshToken string) error {
	claims := &authdomain.RefreshClaims{}
	token, err := jwt.ParseWithClaims(refreshToken, claims, func(token *jwt.Token) (any, error) {
		return []byte(s.cfg.RefreshSecret), nil
	}, jwt.WithAudience(s.cfg.Audience), jwt.WithIssuer(s.cfg.Issuer))
	if err != nil || !token.Valid {
		return errors.New("invalid refresh token")
	}

	return s.refreshStore.Delete(ctx, claims.SessionID)
}

func (s *Service) generateTokenPair(ctx context.Context, userID int64, email, role string) (*authdomain.TokenPair, error) {
	now := time.Now()
	accessExp := now.Add(s.cfg.AccessTokenTTL)
	refreshExp := now.Add(s.cfg.RefreshTokenTTL)
	sessionID, err := randomHex(16)
	if err != nil {
		return nil, err
	}

	accessClaims := authdomain.AccessClaims{
		UserID: userID,
		Email:  email,
		Role:   role,
		Type:   "access",
		RegisteredClaims: jwt.RegisteredClaims{
			Issuer:    s.cfg.Issuer,
			Subject:   toString(userID),
			Audience:  jwt.ClaimStrings{s.cfg.Audience},
			ExpiresAt: jwt.NewNumericDate(accessExp),
			IssuedAt:  jwt.NewNumericDate(now),
			NotBefore: jwt.NewNumericDate(now),
		},
	}

	refreshClaims := authdomain.RefreshClaims{
		UserID:    userID,
		SessionID: sessionID,
		Type:      "refresh",
		RegisteredClaims: jwt.RegisteredClaims{
			Issuer:    s.cfg.Issuer,
			Subject:   toString(userID),
			Audience:  jwt.ClaimStrings{s.cfg.Audience},
			ExpiresAt: jwt.NewNumericDate(refreshExp),
			IssuedAt:  jwt.NewNumericDate(now),
			NotBefore: jwt.NewNumericDate(now),
		},
	}

	accessToken, err := jwt.NewWithClaims(jwt.SigningMethodHS256, accessClaims).SignedString([]byte(s.cfg.AccessSecret))
	if err != nil {
		return nil, err
	}

	refreshToken, err := jwt.NewWithClaims(jwt.SigningMethodHS256, refreshClaims).SignedString([]byte(s.cfg.RefreshSecret))
	if err != nil {
		return nil, err
	}

	if err := s.refreshStore.Save(ctx, sessionID, userID, s.cfg.RefreshTokenTTL); err != nil {
		return nil, err
	}

	return &authdomain.TokenPair{
		AccessToken:  accessToken,
		RefreshToken: refreshToken,
	}, nil
}

func randomHex(n int) (string, error) {
	b := make([]byte, n)
	if _, err := rand.Read(b); err != nil {
		return "", err
	}
	return hex.EncodeToString(b), nil
}

func toString(v int64) string {
	return fmt.Sprintf("%d", v)
}
```

Catatan penting: library `golang-jwt/jwt/v5` punya parsing dan validasi claims lewat `ParseWithClaims`, dan rilis 2026 juga menekankan agar error parsing/validation tidak dicek secara terlalu sempit supaya token invalid tidak salah diterima. Jadi tetap treat hasil parse error sebagai invalid kecuali Anda memang menangani kode error spesifik dengan hati-hati. ([Go Packages][1])

# Refresh token store di Redis

```go id="v605tm"
// File: internal/infra/redis/refresh_store.go
package redis

import (
	"context"
	"fmt"
	"strconv"
	"time"

	goredis "github.com/redis/go-redis/v9"
)

type RefreshStore struct {
	client *goredis.Client
}

func NewRefreshStore(client *goredis.Client) *RefreshStore {
	return &RefreshStore{client: client}
}

func (s *RefreshStore) Save(ctx context.Context, sessionID string, userID int64, ttl time.Duration) error {
	key := fmt.Sprintf("refresh:%s", sessionID)
	return s.client.Set(ctx, key, strconv.FormatInt(userID, 10), ttl).Err()
}

func (s *RefreshStore) Exists(ctx context.Context, sessionID string) (bool, error) {
	key := fmt.Sprintf("refresh:%s", sessionID)
	n, err := s.client.Exists(ctx, key).Result()
	return n > 0, err
}

func (s *RefreshStore) Delete(ctx context.Context, sessionID string) error {
	key := fmt.Sprintf("refresh:%s", sessionID)
	return s.client.Del(ctx, key).Err()
}
```

# JWT middleware untuk Gin

Gin mendukung middleware chaining dengan baik, dan memang itu pola yang paling pas untuk auth guard. ([Gin Web Framework][4])

```go id="qs2534"
// File: internal/app/middleware/auth_jwt.go
package middleware

import (
	"net/http"
	"strings"

	"github.com/gin-gonic/gin"
	authdomain "yourapp/internal/modules/auth/domain"
)

type JWTService interface {
	ParseAccessToken(token string) (*authdomain.AccessClaims, error)
}

func AuthJWT(authSvc JWTService) gin.HandlerFunc {
	return func(c *gin.Context) {
		header := c.GetHeader("Authorization")
		if header == "" {
			c.AbortWithStatusJSON(http.StatusUnauthorized, gin.H{"message": "missing authorization header"})
			return
		}

		parts := strings.SplitN(header, " ", 2)
		if len(parts) != 2 || !strings.EqualFold(parts[0], "Bearer") {
			c.AbortWithStatusJSON(http.StatusUnauthorized, gin.H{"message": "invalid authorization header"})
			return
		}

		claims, err := authSvc.ParseAccessToken(parts[1])
		if err != nil {
			c.AbortWithStatusJSON(http.StatusUnauthorized, gin.H{"message": "invalid or expired token"})
			return
		}

		c.Set("auth.user_id", claims.UserID)
		c.Set("auth.email", claims.Email)
		c.Set("auth.role", claims.Role)

		c.Next()
	}
}
```

```go id="kg703l"
// File: internal/modules/auth/transport/http/handler.go
package http

import (
	"net/http"

	"github.com/gin-gonic/gin"
	authdomain "yourapp/internal/modules/auth/domain"
)

type Handler struct {
	service authdomain.Service
}

func NewHandler(service authdomain.Service) *Handler {
	return &Handler{service: service}
}

func (h *Handler) Login(c *gin.Context) {
	var req LoginRequest
	if err := c.ShouldBindJSON(&req); err != nil {
		c.JSON(http.StatusBadRequest, gin.H{"message": "invalid request"})
		return
	}

	tokenPair, err := h.service.Login(c.Request.Context(), authdomain.LoginInput{
		Email:    req.Email,
		Password: req.Password,
	})
	if err != nil {
		c.JSON(http.StatusUnauthorized, gin.H{"message": "invalid credentials"})
		return
	}

	c.JSON(http.StatusOK, TokenResponse{
		AccessToken:  tokenPair.AccessToken,
		RefreshToken: tokenPair.RefreshToken,
		TokenType:    "Bearer",
	})
}

func (h *Handler) Refresh(c *gin.Context) {
	var req RefreshRequest
	if err := c.ShouldBindJSON(&req); err != nil {
		c.JSON(http.StatusBadRequest, gin.H{"message": "invalid request"})
		return
	}

	tokenPair, err := h.service.Refresh(c.Request.Context(), req.RefreshToken)
	if err != nil {
		c.JSON(http.StatusUnauthorized, gin.H{"message": "invalid refresh token"})
		return
	}

	c.JSON(http.StatusOK, TokenResponse{
		AccessToken:  tokenPair.AccessToken,
		RefreshToken: tokenPair.RefreshToken,
		TokenType:    "Bearer",
	})
}

func (h *Handler) Logout(c *gin.Context) {
	var req RefreshRequest
	if err := c.ShouldBindJSON(&req); err != nil {
		c.JSON(http.StatusBadRequest, gin.H{"message": "invalid request"})
		return
	}

	if err := h.service.Logout(c.Request.Context(), req.RefreshToken); err != nil {
		c.JSON(http.StatusUnauthorized, gin.H{"message": "invalid refresh token"})
		return
	}

	c.JSON(http.StatusOK, gin.H{"message": "logout success"})
}
```

```go id="a12ygz"
	if err != nil {

	return &row, nil
```

# Best practice yang wajib

## 1. Jangan simpan JWT access token di database

Access token harus tetap stateless.

## 2. Simpan refresh token session di Redis

Supaya bisa revoke.

## 3. Pisahkan secret access dan refresh

Jangan satu secret untuk semua.

## 4. Validasi claim penting

Minimal:

* `exp`
* `nbf`
* `iss`
* `aud`
* `type`

Library `golang-jwt/jwt/v5` mendukung validasi parse dengan audience dan issuer options. ([Go Packages][1])

## 5. Untuk high security, pertimbangkan RS256

Kalau nanti auth issuer dipisah dari API verifier, pakai RSA/ECDSA. Package `golang-jwt` mendukung metode signing berbeda dan punya contoh HTTP verification dengan key files. ([GitHub][5])

## 6. Middleware jangan query DB tiap request

Kalau Anda sudah pakai access token berisi `user_id`, `email`, `role`, cukup parse token saja. Query DB hanya jika benar-benar perlu.

## 7. Rate limit endpoint auth

Minimal:

* `/auth/login`
* `/auth/refresh`
* `/auth/logout`

Gin memang menganjurkan middleware sesuai kebutuhan untuk high traffic, dan rate limiting termasuk salah satu concern umum di ekosistem middleware-nya. ([Gin Web Framework][4])

# Apakah perlu pakai `appleboy/gin-jwt`?

Bisa, karena itu middleware JWT populer untuk Gin dan menyediakan login/refresh handler bawaan. ([GitHub][2])

Tapi untuk arsitektur modular seperti punya Anda, saya lebih menyarankan:

**buat middleware JWT sendiri + pakai `golang-jwt/jwt/v5` langsung**

Karena:

* flow lebih jelas
* cocok dengan clean architecture
* lebih mudah custom Redis refresh store
* lebih mudah integrasi dengan domain/service Anda

Pakai `appleboy/gin-jwt` lebih cocok kalau Anda ingin setup cepat, bukan kontrol arsitektur penuh. ([GitHub][2])
